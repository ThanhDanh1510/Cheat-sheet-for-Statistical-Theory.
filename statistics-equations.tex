\sect{Các phân phối thông dụng}
\subsect{Nhị thức $B(n,p)$, $p=K/N$}
\[
 \begin{aligned}
 P(X=k)&=\binom nk p^k(1-p)^{n-k},\quad k=0,1,\ldots,n,\\
 \mathrm E(X)&=np,\qquad \mathrm{Var}(X)=np(1-p).
 \end{aligned}
\]
\[
 B(n,p)\approx N\bigl(np,np(1-p)\bigr)
 \quad(n\geq50,\ np\geq5,\ n(1-p)\geq5).
\]

\subsect{Poisson $P(\mu)$}
\[
 P(X=k)=e^{-\mu}\frac{\mu^k}{k!},\quad k=0,1,\ldots;\qquad
 \mathrm E(X)=\mu,\quad \mathrm{Var}(X)=\mu.
\]

\subsect{Đều rời rạc}
\[
 \begin{aligned}
 P(X=x_i)&=\frac1N,\quad i=1,\ldots,N,\\
 \mathrm E(X)&=\frac1N\sum_{i=1}^{N}x_i,\\
 \mathrm{Var}(X)&=\frac1N\sum_{i=1}^{N}[x_i-\mathrm E(X)]^2.
 \end{aligned}
\]

\subsect{Chuẩn $N(\mu,\sigma^2)$}
\[
 \begin{aligned}
 f_X(x)&=\frac1{\sigma\sqrt{2\pi}}
 \exp\!\left[-\frac{(x-\mu)^2}{2\sigma^2}\right],\quad x\in\mathrm R,\\
 \mathrm E(X)&=\mu,\qquad \mathrm{Var}(X)=\sigma^2.
 \end{aligned}
\]

\subsect{Đều liên tục $U(a,b)$}
\[
 \begin{aligned}
 f_X(x)&=\frac1{b-a},\quad x\in[a,b],\\
 \mathrm E(X)&=\frac{a+b}2,\qquad \mathrm{Var}(X)=\frac{(b-a)^2}{12}.
 \end{aligned}
\]

\subsect{Mũ $\text{Exp}(\lambda)$}
\[
 \begin{aligned}
 f_X(x)&=\lambda e^{-\lambda x},\quad x\ge0,\\
 \mathrm E(X)&=\frac1\lambda,\qquad \mathrm{Var}(X)=\frac1{\lambda^2}.
 \end{aligned}
\]

\subsect{Gamma $\Gamma(\alpha,\beta)$}
\[
 \begin{aligned}
 f_X(x)&=\frac{\beta^\alpha}{\Gamma(\alpha)} x^{\alpha-1} e^{-\beta x},\quad x>0,\\
 \mathrm E(X)&=\frac\alpha\beta,\qquad \mathrm{Var}(X)=\frac\alpha{\beta^2}.
 \end{aligned}
\]

\subsect{Hàm Gamma $\Gamma(\alpha)$}
\[
 \Gamma(\alpha) = \int_0^\infty x^{\alpha-1} e^{-x} \, dx.
\]
Công thức tính nhanh bài tập:
* $\Gamma(\alpha+1) = \alpha\Gamma(\alpha)$
* Với $n$ nguyên dương: $\Gamma(n) = (n-1)!$
* Giá trị đặc biệt: $\Gamma\left(\frac12\right) = \sqrt{\pi}$
* Ví dụ: $\Gamma\left(\frac32\right) = \frac12\sqrt{\pi}$, $\Gamma\left(\frac52\right) = \frac32 \cdot \frac12\sqrt{\pi}$

\subsect{Chi bình phương $\chi^2(n)$}
\[
 \begin{aligned}
 f_Z(z)&=\frac{z^{n/2-1}e^{-z/2}}{2^{n/2}\Gamma(n/2)},\quad z\geq0,\\
 \mathrm E(Z)&=n,\qquad \mathrm{Var}(Z)=2n.
 \end{aligned}
\]

\subsect{Student $St(n)$}
\[
 f_Z(z)=\frac{\Gamma\!\left(\frac{n+1}{2}\right)}
 {\sqrt{n\pi}\,\Gamma\!\left(\frac n2\right)}
 \left(1+\frac{z^2}{n}\right)^{-\frac{n+1}{2}},\quad z\in\mathrm R;
\]
\[
 \mathrm E(Z)=0\ (n>1),\qquad
 \mathrm{Var}(Z)=\frac n{n-2}\ (n>2).
\]

\subsect{Fisher $F(m,n)$}
\[
 f_Z(z)=\frac{\Gamma\!\left(\frac{m+n}{2}\right)
 \left(\frac mn\right)^{m/2}z^{m/2-1}}
 {\Gamma\!\left(\frac m2\right)\Gamma\!\left(\frac n2\right)
 \left(1+\frac mn z\right)^{(m+n)/2}},\quad z>0;
\]
\[
 \mathrm E(Z)=\frac n{n-2}\ (n>2),\qquad
 \mathrm{Var}(Z)=\frac{2n^2(m+n-2)}{m(n-2)^2(n-4)}\ (n>4).
\]


\sect{Phân phối mẫu của trung bình và phương sai}
\textbf{Khi mẫu từ PP chuẩn $X\sim N(\mu,\sigma^2)$:}
\[
 \overline X\perp S^2,\qquad
 \overline X\sim N\!\left(\mu,\frac{\sigma^2}{n}\right),\qquad
 \frac{(n-1)S^2}{\sigma^2}\sim\chi^2(n-1),
\]
\[
 \frac{\overline X-\mu}{S/\sqrt n}\sim T(n-1).
\]
\textbf{Khi mẫu bất kỳ, mẫu lớn $n\geq30$ (theo CLT):}
\[
 \frac{\overline X-\mu}{\sigma/\sqrt n}\approx N(0,1),\qquad
 \frac{\overline X-\mu}{S/\sqrt n}\approx N(0,1).
\]

\sect{1. Ước lượng và tính chất ước lượng}
\textbf{Ước lượng không chệch:} $\mathrm E(\hat\theta)=\theta$.
\quad\textbf{Độ chệch (Bias):}
$B(\hat\theta)=\mathrm E(\hat\theta)-\theta$.
\[
 \mathrm{MSE}(\hat\theta)=\mathrm E[(\hat\theta-\theta)^2]
 =\mathrm{Var}(\hat\theta)+\bigl(\mathrm{Bias}(\hat\theta)\bigr)^2.
\]
$\hat\theta_1$ hiệu quả hơn $\hat\theta_2$ nếu cả hai đều không chệch và
$\mathrm{Var}(\hat\theta_1)<\mathrm{Var}(\hat\theta_2)$.

\textbullet\ Cho trước hai ước lượng, $\hat\theta$ và $\tilde\theta$, tiêu chuẩn
MSE cho phép ta chọn $\hat\theta$ nếu, với cỡ mẫu $n$,
\[
 \mathrm{MSE}(\hat\theta)<\mathrm{MSE}(\tilde\theta),
\]
\textbullet\ hoặc
\[
 \mathrm{Var}(\tilde\theta)-\mathrm{Var}(\hat\theta)>
 \bigl(\mathrm{Bias}(\hat\theta)\bigr)^2-
 \bigl(\mathrm{Bias}(\tilde\theta)\bigr)^2.
\]

\textbf{Ước lượng vững:}
\[
 \lim_{n\to\infty}P\bigl(|\hat\theta_n-\theta|>\epsilon\bigr)=0,
 \qquad \forall\epsilon>0.
\]
\textbf{Mômen tổng thể và mômen mẫu:}
\[
 \mu'_j(\theta)=\mathrm E(X^j)=\int x^j f(x)\,dx,
 \qquad m_j=\frac1n\sum_{i=1}^{n}X_i^j.
\]
\textbf{Định lý Chebyshev:}
\[
 P\bigl(|X-\mathrm E(X)|\geq\epsilon\bigr)
 \leq\frac{\mathrm{Var}(X)}{\epsilon^2}.
\]
\textbf{Hàm hợp lý:}
\[
 L(\theta) = L(\theta \mid X_1, \ldots, X_n) = \prod_{i=1}^{n}f(X_i\mid\theta).
\]
Hàm hợp lý mẫu $X_1,\dots,X_n$ của một số PP phổ biến: \\[2.5pt]
\textbullet\ Bernoulli($p$): $L(p) = p^{\sum X_i}(1-p)^{n-\sum X_i}$ \\[2.5pt]
\textbullet\ Nhị thức $B(k,p)$: $L(p) = \left(\prod\binom{k}{X_i}\right) p^{\sum X_i}(1-p)^{nk-\sum X_i}$ \\[2.5pt]
\textbullet\ Poisson($\lambda$): $L(\lambda) = \frac{e^{-n\lambda}\lambda^{\sum X_i}}{\prod X_i!}$ \\[2.5pt]
\textbullet\ Chuẩn $N(\mu,\sigma^2)$: $L(\mu,\sigma^2) = (2\pi\sigma^2)^{-n/2}\exp\left(-\frac{\sum(X_i-\mu)^2}{2\sigma^2}\right)$ \\[2.5pt]
\textbullet\ Mũ Exp($\lambda$): $L(\lambda) = \lambda^n e^{-\lambda\sum X_i}$

\sect{2. Khoảng tin cậy}
\textbf{Sai số ước lượng $\epsilon$:}
\[
 \epsilon=
 \begin{cases}
 z_{1-\alpha/2}\dfrac{\sigma}{\sqrt n} & \text{(biết $\sigma$)}\\[1.5pt]
 z_{1-\alpha/2}\dfrac{s}{\sqrt n} & \text{($n\geq30$, chưa biết $\sigma$)}\\[1.5pt]
 t^{n-1}_{1-\alpha/2}\dfrac{s}{\sqrt n} & \text{($n<30$, chuẩn, chưa biết $\sigma$)}
 \end{cases}
\]
\[
 \epsilon=z_{1-\alpha/2}\sqrt{\frac{\hat p(1-\hat p)}{n}} \quad\text{(tỷ lệ)}.
\]
\textbf{KTC 2 phía:} $(\overline X-\epsilon, \overline X+\epsilon)$ hoặc $(\hat p-\epsilon, \hat p+\epsilon)$.\\
\textbf{KTC tối đa (chỉ có chặn trên):} $(-\infty, \overline X+\epsilon_{1-\alpha})$ với $\epsilon_{1-\alpha}$ dùng $z_{1-\alpha}$ hoặc $t_{1-\alpha}$.\\
\textbf{KTC tối thiểu (chỉ có chặn dưới):} $(\overline X-\epsilon_{1-\alpha}, +\infty)$.

\sect{3. Kiểm định giả thuyết}

\[
 \beta(\theta)=P(\text{Bác bỏ }H_0\mid\theta)
 =P(\text{Thống kê kiểm định}\in C\mid\theta).
\]
\textbf{Năng lực tại $\theta_1$:} Giá trị hàm năng lực tại điểm cụ thể
$\theta_1\in H_1$, ký hiệu $\beta(\theta_1)=1-\beta$
($\beta$ là xác suất sai lầm loại II).

\textbf{Mức ý nghĩa ($\alpha$):}
\[
 \alpha=\max_{\theta\in H_0}\beta(\theta)
 =P(\text{Bác bỏ }H_0\mid H_0\text{ đúng}).
\]

\sect{6. Hồi quy tuyến tính}
\[
  \begin{aligned}
  S_{xx}&=\sum(x_i-\bar x)^2 =\sum x_i^2-\frac{(\sum x_i)^2}{n},\\
  S_{yy}&=SST=\sum(y_i-\bar y)^2 =\sum y_i^2-\frac{(\sum y_i)^2}{n},\\
  S_{xy}&=\sum(x_i-\bar x)(y_i-\bar y) =\sum x_iy_i-\frac{(\sum x_i)(\sum y_i)}{n}.
  \end{aligned}
\]
\tighttable{\begin{tabular}{c|c}
$SSR = \hat\beta_1^2 S_{xx} = \frac{S_{xy}^2}{S_{xx}}$ & $R^2 = \frac{SSR}{SST} = 1 - \frac{SSE}{SST}$ \\[2.5pt]
$SSE = SST - SSR = \sum(y_i-\hat y_i)^2$ & $MSE = \sigma_{est}^2 = \frac{SSE}{n-2}$ \\[2.5pt]
$\hat\beta_1 = \frac{S_{xy}}{S_{xx}}$ & $\hat\beta_0 = \bar y - \hat\beta_1 \bar x$
\end{tabular}}

\sect{7. Hiệp phương sai và tương quan}
\[
 \begin{aligned}
 \sigma_{XY}&=\mathrm{Cov}(X,Y)\\
 &=\mathrm E[(X-\mathrm E(X))(Y-\mathrm E(Y))]\\
 &=\mathrm E(XY)-\mathrm E(X)\mathrm E(Y).
 \end{aligned}
\]
\[
 \rho_{XY}=\frac{\mathrm{Cov}(X,Y)}{\sigma_X\sigma_Y}
 =\frac{\sigma_{XY}}{\sqrt{\mathrm{Var}(X)\mathrm{Var}(Y)}}.
\]
\[
 r_{xy}=\frac{s_{xy}}{s_xs_y}
 =\frac{S_{xy}}{\sqrt{S_{xx}S_{yy}}}
 =\frac{\sum_{i=1}^{n}(x_i-\bar x)(y_i-\bar y)}
 {\sqrt{\sum_{i=1}^{n}(x_i-\bar x)^2\sum_{i=1}^{n}(y_i-\bar y)^2}}.
\]
\textbf{Kiểm định hệ số tương quan ($H_0:\rho_{XY}=0$):}
\[
 T_0=\frac{r_{xy}\sqrt{n-2}}{\sqrt{1-r_{xy}^2}}\sim T(n-2).
\]
Bác bỏ $H_0$ nếu $|t_0|>t^{n-2}_{1-\alpha/2}$ (với $H_1:\rho_{XY}\ne 0$).

\sect{8. Kiểm định tính độc lập (Chi bình phương)}
Bảng ngẫu nhiên $r \times s$. Tần số thực nghiệm $n_{ij}$, tổng hàng $n_i$, tổng cột $m_j$, cỡ mẫu $N$. \\
Tần số lý thuyết: $e_{ij} = \frac{n_i m_j}{N}$ (điều kiện $e_{ij} \geq 5$). \\
Giả thuyết $H_0$: Hai đặc tính độc lập. \\
Thống kê kiểm định:
\[
 Q^2 = \sum_{i=1}^{r}\sum_{j=1}^{s} \frac{(n_{ij}-e_{ij})^2}{e_{ij}} = \sum_{i=1}^{r}\sum_{j=1}^{s} \frac{n_{ij}^2}{e_{ij}} - N.
\]
Bác bỏ $H_0$ nếu $Q^2 > \chi^2_{(r-1)(s-1)}(\alpha)$ (với mức ý nghĩa $\alpha$).
